import { Component, inject, signal } from '@angular/core'; import { RouterLink } from '@angular/router'; @Component({ selector: 'app-publisher.component', imports: [], templateUrl: './publisher.component.html', styleUrl: './publisher.component.css' }) export class PublisherComponent { private svc = inject(AutoresService); private auth = inject(AuthService); //Ver o token autores = signal([]); carregando = signal(true); erro = signal(null); constructor() { console.log("Token de acesso: ", this.auth.token()); this.svc.listar().subscribe({ next: (data) => { this.autores.set(data); this.carregando.set(false); }, error: () => { this.erro.set('Falha ao carregar autores'); this.carregando.set(false); } }); } }
